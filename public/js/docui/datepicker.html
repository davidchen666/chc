$p.plugin.datePicker('#my_picker', {
    textFormat: "YYYY年MM月DD日",
    defaultDate: moment(),
    //minViewMode:'months' //按照月显示，或是使用days，按照天显示。
}, function(date) {
    $p.com.alert(date);
});
