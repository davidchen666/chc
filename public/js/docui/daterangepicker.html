Javascript代码


$p.plugin.dateRangePicker("#my_picker", {
    textForamt: "YYYY年MM月DD日",
    defaultDate:[moment().subtract('days', 6),moment()],  //默认最近一周
}, function(start, end) {
    $p.com.alert("开始时间：" + start + " , 结束时间：" + end);
});
