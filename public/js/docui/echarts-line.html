require('../../plugins/echarts/module');

var chart = $p.plugin.echarts("my_chart", {
    type: "line",
    title: {
        text: '来源分析',
        subtext: '最近一周数据'
    }
});

chart.load({
    column: ['2014-07-05', '2014-07-06', '2014-07-07', '2014-07-08', '2014-07-09', '2014-07-10', '2014-07-11'],
