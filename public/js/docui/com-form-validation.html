require('../../plugins/jquery.validate/module');

$p.com.form("#form_sample_1", {
    validate: {
        rules: {
            name: {
                minlength: 2,
                required: true
            },
            email: {
                required: true,
                email: true
            },
            url: {
                required: true,
                url: true
            },
            number: {
                required: true,
                number: true
            },
            digits: {
                required: true,
                digits: true
            },
            creditcard: {
                required: true,
                creditcard: true
            },
            occupation: {
                minlength: 5,
            },
            category: {
                required: true
            }
        },
        custom: function(form, data) {
            return true;
        }
    },
    submit: function(form, data) {
        model.add(data, function(resp) {
            $p.com.alert(resp.message);
        })
    }
});
